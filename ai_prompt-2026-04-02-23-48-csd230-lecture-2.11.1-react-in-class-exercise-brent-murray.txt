import { useState, useEffect } from 'react'
import { Routes, Route } from 'react-router'
import Navbar from './Navbar'
import Home from './Home'
import Book from './Book'
import BookForm from './BookForm'
import Magazine from './Magazine'
import MagazineForm from './MagazineForm'
import './App.css'

function App() {
    const [books, setBooks] = useState([]);
    const [magazines, setMagazines] = useState([]);
    const [loading, setLoading] = useState(true);

    useEffect(() => {
        // Fetch Books and Magazines simultaneously
        Promise.all([
            fetch('/api/books').then(res => res.json()),
            fetch('/api/magazines').then(res => res.json())
        ]).then(([booksData, magazinesData]) => {
            setBooks(booksData);
            setMagazines(magazinesData);
            setLoading(false);
        });
    }, []);

    // --- Book Handlers ---
    const handleAddBook = (newBook) => setBooks([...books, newBook]);

    const handleDeleteBook = (id) => {
        if (!window.confirm("Delete this book?")) return;
        fetch(`/api/books/${id}`, { method: 'DELETE' })
            .then(res => { if (res.ok) setBooks(books.filter(b => b.id !== id)); });
    };

    const handleUpdateBook = (id, updatedData) => {
        fetch(`/api/books/${id}`, {
            method: 'PUT',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(updatedData)
        })
            .then(res => res.json())
            .then(savedBook => setBooks(books.map(b => (b.id === id ? savedBook : b))));
    };

    // --- Magazine Handlers ---
    const handleAddMagazine = (newMagazine) => setMagazines([...magazines, newMagazine]);

    const handleDeleteMagazine = (id) => {
        if (!window.confirm("Delete this magazine?")) return;
        fetch(`/api/magazines/${id}`, { method: 'DELETE' })
            .then(res => { if (res.ok) setMagazines(magazines.filter(m => m.id !== id)); });
    };

    const handleUpdateMagazine = (id, updatedData) => {
        // Ensure Spring gets seconds for LocalDateTime parsing
        if (updatedData.currentIssue && updatedData.currentIssue.length === 16) {
            updatedData.currentIssue += ':00';
        }

        fetch(`/api/magazines/${id}`, {
            method: 'PUT',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(updatedData)
        })
            .then(res => res.json())
            .then(savedMagazine => setMagazines(magazines.map(m => (m.id === id ? savedMagazine : m))));
    };

    if (loading) return <h2>Loading...</h2>;

    return (
        <div className="app-container" style={{ maxWidth: '900px', margin: '0 auto', padding: '20px' }}>
            <Navbar />

            <Routes>
                <Route path="/" element={<Home />} />

                {/* Book Routes */}
                <Route path="/inventory" element={
                    <div className="book-list">
                        <h1>Current Book Inventory</h1>
                        {books.map((b) => (
                            <Book key={b.id} {...b} onDelete={handleDeleteBook} onUpdate={handleUpdateBook} />
                        ))}
                    </div>
                } />
                <Route path="/add" element={
                    <div>
                        <h1>Add Book to Library</h1>
                        <BookForm onBookAdded={handleAddBook} />
                    </div>
                } />

                {/* Magazine Routes */}
                <Route path="/magazines" element={
                    <div className="magazine-list">
                        <h1>Current Magazine Inventory</h1>
                        {magazines.map((m) => (
                            <Magazine key={m.id} {...m} onDelete={handleDeleteMagazine} onUpdate={handleUpdateMagazine} />
                        ))}
                    </div>
                } />
                <Route path="/add-magazine" element={
                    <div>
                        <h1>Add Magazine to Library</h1>
                        <MagazineForm onMagazineAdded={handleAddMagazine} />
                    </div>
                } />
            </Routes>
        </div>
    )
}

export default App

This is the updated **README.md** for **Version 4**. It merges the detailed technical explanations from V3 with the new architectural features of V4, and includes a dedicated section highlighting the evolution between versions.

***

# Generic Prompt Generator V4

**Generic Prompt Generator V4** is a Spring Boot-based CLI utility that recursively scans a local directory, filters files based on customizable rules (similar to `.gitignore`), and consolidates the content of accepted files into a single, well-structured text file.

The resulting output is optimized for providing comprehensive project context to Large Language Models (LLMs) like ChatGPT, Claude, or Gemini.

## 🚀 What's New in Version 4?

Version 4 introduces several "AI-Safety" and "Architectural Awareness" features that were missing in previous versions:
*   **Project Structure Map:** Automatically generates a visual directory tree at the top of the output file so the LLM understands your file hierarchy before reading the code.
*   **Enhanced Stack Support:** Default ignore rules are now pre-configured for **Spring Boot, Flutter, Maven, Gradle, IntelliJ, NetBeans, Eclipse, and VS Code**.
*   **Context Protection (Size Limits):** Automatically skips individual files larger than **250KB** to prevent "context stuffing" where a single large file (like a log or JSON dump) exhausts the LLM's memory.
*   **Token Estimation:** Provides a rough token count estimate in the console upon completion.
*   **Improved Maven/Gradle Handling:** Automatically ignores wrapper scripts and binaries (`mvnw`, `gradlew`, `.mvn/`) to keep the focus on logic.

---

## 🔄 Differences: V3 vs. V4

| Feature | Version 3 | Version 4 |
| :--- | :--- | :--- |
| **Output Format** | Linear file list | **Visual Tree Map** + File list |
| **Size Management** | None (reads everything) | **250KB per-file ceiling** |
| **Default Ignores** | Basic Java/Node | **Full Stack (Flutter, Maven, 4+ IDEs)** |
| **Infrastructure** | Standard recursive walk | **Tree-aware traversal** |
| **Console Feedback** | Simple logging | **Size & Token Estimation** |
| **Safety** | Binary detection | **Binary detection + Size filtering** |

---

## ✨ Core Features

*   **Recursive Scanning:** Walks through the target directory tree regardless of depth.
*   **Smart Filtering:** Uses a `.files2ignore` file to exclude noise.
*   **Binary Detection:** Automatically detects and skips binary files (images, PDFs, compiled artifacts) using NULL-byte inspection.
*   **Auto-Initialization:** Generates a robust, modern `.files2ignore` if one is missing.
*   **Timestamped Output:** Prevents overwriting previous runs (e.g., `ai_prompt-2026-02-14-08-00-my_project.txt`).

---

## 📖 Usage

### Prerequisites
*   Java 17 or higher (Supports Java 21)
*   Maven

### Building
```bash
./mvnw clean package
```

### Running
Provide the path of the project you wish to scan:
```bash
java -jar target/generic_prompt_generator_v4-0.0.1-SNAPSHOT.jar /path/to/project
```

---

## ⚙️ Configuration: `.files2ignore`

The tool uses standard "glob" pattern matching. If no file exists, V4 creates one with specific ignores for:
*   **Build Tools:** `target/`, `build/`, `.gradle/`, `.mvn/`, `mvnw`, `gradlew`
*   **IDEs:** `.idea/`, `.vscode/`, `.settings/`, `nbproject/`
*   **Mobile:** `.dart_tool/`, `ios/Pods/`, `*.g.dart`
*   **System:** `.git/`, `.DS_Store`, `Thumbs.db`, `*.log`

### How Rules Apply (Recursive Logic)

1.  **Standard Rules:** Rules without a leading slash apply everywhere.
   *   `target/` matches `/target` and `/subdir/target`.
2.  **Root-Only Rules:** Rules starting with `/` are anchored to the root.
   *   `/README.md` matches the root file but **not** `src/docs/README.md`.
3.  **Negations:** Use `!` to force inclusion.
   *   `!important.log` will include that file even if `*.log` is ignored.

---

## 🛡️ Safety & Limits

*   **Max File Size:** Any file exceeding **250KB** is skipped. This prevents the prompt from becoming too large for the LLM to process and saves on API costs/token usage.
*   **Encoding:** The tool strictly uses `UTF-8`. If a file uses an incompatible encoding, it will insert an `[ERROR: Invalid UTF-8 Content]` placeholder rather than crashing.
*   **Binary Safety:** The tool reads the first block of every file; if a `NULL` byte is detected, the file is treated as binary and skipped.

package com.example.generic_prompt_generator_v4;

import org.springframework.boot.CommandLineRunner;
import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;

import java.io.BufferedWriter;
import java.io.IOException;
import java.io.InputStream;
import java.nio.charset.MalformedInputException;
import java.nio.charset.StandardCharsets;
import java.nio.file.*;
import java.nio.file.attribute.BasicFileAttributes;
import java.time.LocalDateTime;
import java.time.format.DateTimeFormatter;
import java.util.ArrayList;
import java.util.List;

@SpringBootApplication
public class GenericPromptGeneratorV4Application implements CommandLineRunner {

    private static final String IGNORE_FILE_NAME = ".files2ignore";
    private static final long MAX_FILE_SIZE_BYTES = 250 * 1024; // 250KB limit per file

    private static final String DEFAULT_IGNORE_CONTENT = """
            # .files2ignore GenericPromptGeneratorV4
            ai_prompt*
            Timeline*.json
            assets/
            
            # --- IDEs ---
            .idea/
            *.iml
            *.iws
            *.ipr
            out/
            .settings/
            .metadata/
            .classpath
            .project
            .factorypath
            .springBeans
            nbproject/
            nbbuild/
            dist/
            nbdist/
            .nb-gradle/
            .vscode/
            
            # --- Build & Runtime ---
            target/
            .mvn/
            mvnw
            mvnw.cmd
            .gradle/
            build/
            gradlew
            gradlew.bat
            gradle/wrapper/
            node_modules/
            
            # --- Databases / Volumes (Common permission issues) ---
            postgres-data/
            mysql-data/
            data/db/
            
            # --- Flutter / Dart ---
            .dart_tool/
            .packages
            .pub-cache/
            .pub/
            ios/Pods/
            android/.gradle/
            *.g.dart
            *.freezed.dart
            
            # --- OS & Misc ---
            .git/
            .DS_Store
            Thumbs.db
            *.log
            *.tmp
            *.bak
            
            # --- Binaries & Archives ---
            *.class
            *.jar
            *.war
            *.ear
            *.zip
            *.tar.gz
            *.rar
            *.exe
            *.dll
            *.so
            *.dylib
            """;

    public static void main(String[] args) {
        SpringApplication.run(GenericPromptGeneratorV4Application.class, args);
    }

    @Override
    public void run(String... args) throws Exception {
        System.out.println("========================================");
        System.out.println("Generic Prompt Generator V4 (Safety-First)");
        System.out.println("========================================");

        if (args.length == 0) {
            System.err.println("Usage: java -jar app.jar <folder-path>");
            return;
        }

        Path rootPath = Paths.get(args[0]).toAbsolutePath().normalize();
        if (!Files.exists(rootPath)) {
            System.err.println("Error: Path does not exist: " + rootPath);
            return;
        }

        Path ignoreFile = rootPath.resolve(IGNORE_FILE_NAME);
        if (!Files.exists(ignoreFile)) {
            Files.writeString(ignoreFile, DEFAULT_IGNORE_CONTENT);
            System.out.println("Generated default .files2ignore");
        }
        List<IgnoreRule> rules = parseIgnoreRules(ignoreFile);

        String timestamp = LocalDateTime.now().format(DateTimeFormatter.ofPattern("yyyy-MM-dd-HH-mm"));
        String outputFileName = "ai_prompt-" + timestamp + "-" + rootPath.getFileName() + ".txt";
        Path outputPath = rootPath.resolve(outputFileName);

        try (BufferedWriter writer = Files.newBufferedWriter(outputPath, StandardCharsets.UTF_8)) {
            writer.write("CONTEXT FOR AI MODEL\n");
            writer.write("Generated from: " + rootPath + "\n");
            writer.write("Timestamp: " + timestamp + "\n\n");

            writer.write("--- PROJECT STRUCTURE ---\n");
            generateDirectoryTree(rootPath, rootPath, rules, writer, "");
            writer.write("\n--------------------------------------------------\n\n");

            Files.walkFileTree(rootPath, new SimpleFileVisitor<>() {
                @Override
                public FileVisitResult preVisitDirectory(Path dir, BasicFileAttributes attrs) {
                    if (shouldIgnore(dir, rootPath, rules, true)) return FileVisitResult.SKIP_SUBTREE;
                    return FileVisitResult.CONTINUE;
                }

                @Override
                public FileVisitResult visitFile(Path file, BasicFileAttributes attrs) {
                    if (file.equals(outputPath) || file.equals(ignoreFile)) return FileVisitResult.CONTINUE;
                    if (shouldIgnore(file, rootPath, rules, false)) return FileVisitResult.CONTINUE;

                    try {
                        if (attrs.size() > MAX_FILE_SIZE_BYTES) {
                            System.out.println("[SKIPPING LARGE] " + rootPath.relativize(file));
                        } else if (isBinaryFile(file)) {
                            System.out.println("[SKIPPING BINARY] " + rootPath.relativize(file));
                        } else {
                            System.out.println("[ADDING] " + rootPath.relativize(file));
                            appendFileContent(writer, rootPath, file);
                        }
                    } catch (IOException e) {
                        System.err.println("[READ ERROR] Skipping " + file + ": " + e.getMessage());
                    }
                    return FileVisitResult.CONTINUE;
                }

                @Override
                public FileVisitResult visitFileFailed(Path file, IOException exc) {
                    System.err.println("[ACCESS DENIED] Skipping " + file);
                    return FileVisitResult.CONTINUE;
                }
            });
        }

        long finalSize = Files.size(outputPath);
        System.out.println("----------------------------------------");
        System.out.println("Success! Created: " + outputFileName);
        System.out.println("Output Size: " + (finalSize / 1024) + " KB");
        System.out.println("Est. Tokens: ~" + (finalSize / 4));
        System.out.println("----------------------------------------");
    }

    private void generateDirectoryTree(Path current, Path root, List<IgnoreRule> rules, BufferedWriter writer, String indent) throws IOException {
        String name = current.getFileName().toString();
        writer.write(indent + "|-- " + name + (Files.isDirectory(current) ? "/" : "") + "\n");

        if (Files.isDirectory(current)) {
            // Using a try-catch specifically for the directory stream to handle Access Denied
            try (DirectoryStream<Path> stream = Files.newDirectoryStream(current)) {
                for (Path entry : stream) {
                    if (!shouldIgnore(entry, root, rules, Files.isDirectory(entry))) {
                        generateDirectoryTree(entry, root, rules, writer, indent + "    ");
                    }
                }
            } catch (AccessDeniedException | SecurityException e) {
                writer.write(indent + "    |-- [ACCESS DENIED]\n");
            }
        }
    }

    private void appendFileContent(BufferedWriter writer, Path root, Path file) throws IOException {
        String relativePath = root.relativize(file).toString().replace("\\", "/");
        writer.write("--- FILE: " + relativePath + " ---\n");
        writer.write("```\n");
        try {
            writer.write(Files.readString(file, StandardCharsets.UTF_8));
        } catch (MalformedInputException e) {
            writer.write("[ERROR: Invalid UTF-8 Content]");
        }
        writer.write("\n```\n\n");
    }

    private boolean isBinaryFile(Path path) throws IOException {
        try (InputStream in = Files.newInputStream(path)) {
            byte[] buffer = new byte[4096];
            int read = in.read(buffer);
            for (int i = 0; i < read; i++) {
                if (buffer[i] == 0) return true;
            }
        } catch (AccessDeniedException e) {
            return true; // Treat as binary/ignore if we can't read it
        }
        return false;
    }

    private boolean shouldIgnore(Path path, Path root, List<IgnoreRule> rules, boolean isDir) {
        Path rel = root.relativize(path);
        if (rel.toString().isEmpty()) return false;
        boolean ignored = false;
        for (IgnoreRule rule : rules) {
            if (rule.dirOnly && !isDir) continue;
            if (rule.matches(rel)) ignored = !rule.isNegation;
        }
        return ignored;
    }

    private List<IgnoreRule> parseIgnoreRules(Path ignoreFile) throws IOException {
        List<IgnoreRule> rules = new ArrayList<>();
        for (String line : Files.readAllLines(ignoreFile)) {
            String t = line.trim();
            if (t.isEmpty() || t.startsWith("#")) continue;
            rules.add(new IgnoreRule(t));
        }
        return rules;
    }

    private static class IgnoreRule {
        final List<PathMatcher> matchers = new ArrayList<>();
        final boolean isNegation;
        final boolean dirOnly;

        IgnoreRule(String pattern) {
            if (pattern.startsWith("!")) { isNegation = true; pattern = pattern.substring(1); }
            else isNegation = false;

            if (pattern.endsWith("/")) { dirOnly = true; pattern = pattern.substring(0, pattern.length()-1); }
            else dirOnly = false;

            if (pattern.startsWith("/")) {
                matchers.add(FileSystems.getDefault().getPathMatcher("glob:" + pattern.substring(1)));
            } else {
                matchers.add(FileSystems.getDefault().getPathMatcher("glob:" + pattern));
                matchers.add(FileSystems.getDefault().getPathMatcher("glob:**/" + pattern));
            }
        }
        boolean matches(Path p) {
            return matchers.stream().anyMatch(m -> m.matches(p));
        }
    }
}

    private final BookRepository bookRepository;

    public BookController(BookRepository bookRepository) {
        this.bookRepository = bookRepository;
    }

    // GET all books
    @GetMapping
    public List<BookEntity> getAllBooks() {
        return bookRepository.findAll();
    }

    // GET single book
    @GetMapping("/{id}")
    public ResponseEntity<BookEntity> getBookById(@PathVariable Long id) {
        return bookRepository.findById(id)
                .map(ResponseEntity::ok)
                .orElse(ResponseEntity.notFound().build());
    }

    @PostMapping
    public BookEntity createBook(@RequestBody BookEntity book) {
        return bookRepository.save(book);
    }

package csd230.seeder;

import csd230.entities.*;
import csd230.repositories.BookRepository;
import csd230.repositories.MagazineRepository;
import net.datafaker.Faker;
import org.springframework.boot.CommandLineRunner;
import org.springframework.stereotype.Component;

import java.time.LocalDateTime;
import java.time.ZoneId;
import java.util.concurrent.TimeUnit;

@Component
public class DataSeeder implements CommandLineRunner {

    private final BookRepository bookRepository;
    private final MagazineRepository magazineRepository;
    private final Faker faker;

    public DataSeeder(BookRepository bookRepository, MagazineRepository magazineRepository) {
        this.bookRepository = bookRepository;
        this.magazineRepository = magazineRepository;
        this.faker = new Faker();
    }

    @Override
    public void run(String... args) throws Exception {
        // Only seed if the database is empty
        if (bookRepository.count() == 0) {
            seedBooks();
            seedMagazines();
        }
    }

    private void seedBooks() {
        System.out.println("Seeding Books...");
        for (int i = 0; i < 10; i++) {
            BookEntity book = new BookEntity(
                    faker.book().title(),                // Title
                    faker.number().randomDouble(2, 10, 100), // Price
                    faker.number().numberBetween(1, 50), // Copies
                    faker.book().author()                // Author
            );
            bookRepository.save(book);
        }
    }

    private void seedMagazines() {
        System.out.println("Seeding Magazines...");
        for (int i = 0; i < 5; i++) {
            // Convert Faker Date to LocalDateTime
            LocalDateTime issueDate = faker.date().past(365, TimeUnit.DAYS)
                    .toInstant().atZone(ZoneId.systemDefault()).toLocalDateTime();

            MagazineEntity mag = new MagazineEntity(
                    faker.book().publisher() + " Weekly", // Using publisher as magazine title
                    faker.number().randomDouble(2, 5, 20),
                    faker.number().numberBetween(10, 100),
                    faker.number().numberBetween(100, 500), // Order Qty
                    issueDate
            );
            magazineRepository.save(mag);
        }
    }
}
